document.onkeydown = checkKey;

function checkKey(e) {
  e = e || window.event;
  if (e.keyCode == '38') {
    e.preventDefault();
    direction = "up";
  }
  else if (e.keyCode == '40') {
    e.preventDefault();
    direction = "down";
  }
  else if (e.keyCode == '37') {
    e.preventDefault();
    direction = "left";
  }
  else if (e.keyCode == '39') {
    e.preventDefault();
    direction = "right";
  }
}
